declare
    cursor cur_tb1 is select * from tb1 for update;
    v_rec tb1%rowtype;
begin
    open cur_tb1;
    loop
        fetch cur_tb1 into v_rec;
        update tb1 set col1 = v_rec.col1 where current of cur_tb1;
        exit when cur_tb1%notfound;
    end loop;
    commit;
    close cur_tb1;
end;

declare
    cursor cur_tb1 is select * from tb1;
    type typ_tb1 is table of tb1%rowtype;
    v_rec typ_tb1;
begin
    open cur_tb1;
    loop
        if mod(cur_tb1%rowcount, 5000) = 0 then
            dbms_output.putline(cur_tb1%rowcount);
        end if; 
        fetch cur_tb1 bullk collect into v_rec limit 2000;
        for I in 1 ... v_rec.count loop
            dosth(v_rec(I));
        end loop;
        commit;
        exit when cur_tb1%notfound;
    end loop;
    close cur_tb1;
end;
